# .github/workflows/ci-cd.yml
# Finance Monitoring AI System — Full CI/CD Pipeline
#
# Branch strategy:
#   feat/* / fix/*  →  PR into develop  →  staging preview
#   develop → main  →  production deploy (Supabase migrations → Render → Vercel)

name: CI / CD

on:
  push:
    branches: [main, develop]
  pull_request:
    branches: [main, develop]

concurrency:
  group: ${{ github.workflow }}-${{ github.ref }}
  cancel-in-progress: true

env:
  PYTHON_VERSION: "3.11"
  NODE_VERSION:   "20"

# ─────────────────────────────────────────────────────────────────────────────
jobs:

  # ── Job 1: Backend lint + test ──────────────────────────────────────────────
  backend:
    name: Backend — Lint & Test
    runs-on: ubuntu-latest

    services:
      postgres:
        image: postgres:15-alpine
        env:
          POSTGRES_USER: postgres
          POSTGRES_PASSWORD: postgres
          POSTGRES_DB: finance_test
        options: >-
          --health-cmd pg_isready
          --health-interval 5s
          --health-timeout 5s
          --health-retries 10
        ports:
          - 5432:5432

    steps:
      - uses: actions/checkout@v4

      - name: Set up Python ${{ env.PYTHON_VERSION }}
        uses: actions/setup-python@v5
        with:
          python-version: ${{ env.PYTHON_VERSION }}
          cache: pip
          cache-dependency-path: backend/requirements*.txt

      - name: Install dependencies
        working-directory: backend
        run: |
          pip install --upgrade pip
          pip install -r requirements.txt -r requirements-dev.txt

      - name: Apply migrations to test DB
        working-directory: backend
        env:
          PGPASSWORD: postgres
        run: |
          psql -h localhost -U postgres -d finance_test \
            -f ../supabase/migrations/001_review_queue_and_duplicates.sql

      - name: Ruff lint
        working-directory: backend
        run: ruff check . --output-format=github

      - name: Run tests with coverage
        working-directory: backend
        env:
          DATABASE_URL:        postgresql://postgres:postgres@localhost:5432/finance_test
          GEMINI_API_KEY:      ${{ secrets.GEMINI_API_KEY }}
          SUPABASE_URL:        ${{ secrets.SUPABASE_URL }}
          SUPABASE_SERVICE_KEY: ${{ secrets.SUPABASE_SERVICE_KEY }}
          SECRET_KEY:          ci-test-secret-key-not-real
        run: |
          pytest --tb=short -q --cov=app --cov-report=xml

      - name: Upload coverage
        uses: codecov/codecov-action@v4
        with:
          file: backend/coverage.xml
          flags: backend

  # ── Job 2: Frontend lint + type-check + test ────────────────────────────────
  frontend:
    name: Frontend — Lint, Types & Test
    runs-on: ubuntu-latest

    steps:
      - uses: actions/checkout@v4

      - name: Set up Node ${{ env.NODE_VERSION }}
        uses: actions/setup-node@v4
        with:
          node-version: ${{ env.NODE_VERSION }}
          cache: npm
          cache-dependency-path: frontend/package-lock.json

      - name: Install dependencies
        working-directory: frontend
        run: npm ci

      - name: ESLint
        working-directory: frontend
        run: npm run lint

      - name: TypeScript type-check
        working-directory: frontend
        run: npx tsc --noEmit

      - name: Unit tests
        working-directory: frontend
        env:
          NEXT_PUBLIC_API_URL:            http://localhost:8000
          NEXT_PUBLIC_SUPABASE_URL:       https://placeholder.supabase.co
          NEXT_PUBLIC_SUPABASE_ANON_KEY:  placeholder
        run: npm test -- --watchAll=false --passWithNoTests --ci

  # ── Job 3: Security scan ────────────────────────────────────────────────────
  security:
    name: Security Scan (Snyk)
    runs-on: ubuntu-latest
    needs: [backend, frontend]
    # Only run on PRs targeting main, or direct pushes to develop/main
    if: github.event_name == 'push' || github.base_ref == 'main'

    steps:
      - uses: actions/checkout@v4

      - name: Snyk Python scan
        uses: snyk/actions/python@master
        with:
          args: --file=backend/requirements.txt --severity-threshold=high
        env:
          SNYK_TOKEN: ${{ secrets.SNYK_TOKEN }}
        continue-on-error: true   # don't block on Snyk outage

      - name: Snyk Node scan
        uses: snyk/actions/node@master
        with:
          args: --file=frontend/package.json --severity-threshold=high
        env:
          SNYK_TOKEN: ${{ secrets.SNYK_TOKEN }}
        continue-on-error: true

  # ── Job 4: Staging deploy (develop branch) ──────────────────────────────────
  deploy-staging:
    name: Deploy → Staging
    runs-on: ubuntu-latest
    needs: [backend, frontend, security]
    if: github.ref == 'refs/heads/develop' && github.event_name == 'push'
    environment: staging

    steps:
      - uses: actions/checkout@v4

      - name: Deploy Next.js preview to Vercel
        env:
          VERCEL_TOKEN:      ${{ secrets.VERCEL_TOKEN }}
          VERCEL_ORG_ID:     ${{ secrets.VERCEL_ORG_ID }}
          VERCEL_PROJECT_ID: ${{ secrets.VERCEL_PROJECT_ID }}
        run: |
          npm i -g vercel --silent
          vercel --token $VERCEL_TOKEN --yes 2>&1 | tail -5

      - name: Notify Slack — staging deployed
        if: always()
        uses: slackapi/slack-github-action@v1.26.0
        with:
          payload: |
            {"text": "🚀 *Staging deploy* `${{ github.ref_name }}` — ${{ job.status }}\n${{ github.event.head_commit.message }}"}
        env:
          SLACK_WEBHOOK_URL: ${{ secrets.SLACK_WEBHOOK_URL }}

  # ── Job 5: Production deploy (main branch) ──────────────────────────────────
  deploy-production:
    name: Deploy → Production
    runs-on: ubuntu-latest
    needs: [backend, frontend, security]
    if: github.ref == 'refs/heads/main' && github.event_name == 'push'
    environment: production

    steps:
      - uses: actions/checkout@v4

      # ── Step A: Run DB migrations FIRST (critical ordering) ──
      - name: Install Supabase CLI
        run: |
          curl -fsSL \
            https://github.com/supabase/cli/releases/latest/download/supabase_linux_amd64.tar.gz \
            | tar -xz -C /usr/local/bin supabase

      - name: Run Supabase migrations
        env:
          SUPABASE_ACCESS_TOKEN: ${{ secrets.SUPABASE_ACCESS_TOKEN }}
          SUPABASE_DB_PASSWORD:  ${{ secrets.SUPABASE_DB_PASSWORD }}
          SUPABASE_PROJECT_ID:   ${{ secrets.SUPABASE_PROJECT_ID }}
        run: |
          supabase link --project-ref $SUPABASE_PROJECT_ID
          supabase db push
          echo "✅ Migrations applied"

      # ── Step B: Deploy FastAPI to Render ──
      - name: Trigger Render deploy (FastAPI)
        env:
          RENDER_API_KEY:    ${{ secrets.RENDER_API_KEY }}
          RENDER_SERVICE_ID: ${{ secrets.RENDER_SERVICE_ID }}
        run: |
          DEPLOY=$(curl -sf -X POST \
            "https://api.render.com/v1/services/${RENDER_SERVICE_ID}/deploys" \
            -H "Authorization: Bearer ${RENDER_API_KEY}" \
            -H "Content-Type: application/json" \
            -d '{"clearCache":false}')
          echo "Render deploy ID: $(echo $DEPLOY | python3 -c 'import sys,json; print(json.load(sys.stdin)["id"])')"

      # ── Step C: Deploy Next.js to Vercel ──
      - name: Deploy Next.js to Vercel (prod)
        env:
          VERCEL_TOKEN:      ${{ secrets.VERCEL_TOKEN }}
          VERCEL_ORG_ID:     ${{ secrets.VERCEL_ORG_ID }}
          VERCEL_PROJECT_ID: ${{ secrets.VERCEL_PROJECT_ID }}
        run: |
          npm i -g vercel --silent
          vercel --token $VERCEL_TOKEN --prod --yes 2>&1 | tail -5

      # ── Step D: Smoke test ──
      - name: Wait for Render + smoke test /health
        run: |
          echo "Waiting 30s for Render to spin up..."
          sleep 30
          curl --fail --retry 5 --retry-delay 10 \
            https://finance-monitor-api.onrender.com/health \
            && echo "✅ Health check passed" \
            || (echo "❌ Health check failed" && exit 1)

      # ── Step E: Notify ──
      - name: Notify Slack — production deployed
        if: always()
        uses: slackapi/slack-github-action@v1.26.0
        with:
          payload: |
            {"text": "${{ job.status == 'success' && '✅' || '❌' }} *Production deploy* — ${{ job.status }}\nCommit: `${{ github.sha }}` by @${{ github.actor }}"}
        env:
          SLACK_WEBHOOK_URL: ${{ secrets.SLACK_WEBHOOK_URL }}
